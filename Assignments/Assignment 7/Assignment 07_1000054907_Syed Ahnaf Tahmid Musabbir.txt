public class Player{
  public static int total;
  public static String playerName [] = new String [11];
  public String name;
  public String playerCountry;
  public int jersey;
  public static int count;
  
  public Player(String s, String t, int n){
    name = s;
    playerName[count] = s;
    playerCountry = t;
    jersey = n;
    this.total++;
    count++;
  }
  
  public String player_detail(){
    return("Player Name: " + name + "\n" + "Jersey Number: " + jersey + "\n" + "Country: " + playerCountry);
  }
  public static void info(){
    System.out.println("Total Number of players: " + count);
    if(count>0){
      System.out.print("Players enlisted so far: " );
      for(int i = 0; i<count; i++){
        System.out.print(playerName[i] + ", ");
      }
    }
    System.out.println();
  }
}

public class Travel{
  private static int count;
  private String source;
  private String destination;
  private int time = 1;
  
  public Travel(String s, String t){
    source = s;
    destination = t;
    count++;
  }
  public void setTime(int n){
    time = n;
  }
  public void setSource(String s){
    source = s;
  }
  public void setDestination(String s){
    destination = s;
  }
  public void setCount(int n){
    count+=n;
  }
  public static int getCount(){
    return count;
  }
  public String getSource(){
    return source;
  }
  public String getDestination(){
    return destination;
  }
  public int getTime(){
    return time;
  }
  public String displayTravelInfo(){
    return("Source: " + getSource() + "\n" + "Destination: " + getDestination() + "\n" + "Flight Time: " + getTime() + ":00");
  }
}

public class Cargo{
  public boolean loaded = false;
  public static int id = 1;
  public int contentID;
  public String contents;
  public double weight;
  public static double capacity = 10.0;
  
  public Cargo(String s, double d){
    contents = s;
    weight = d;
    contentID = id;
    id++;
  }
  public void load(){
    if((capacity - weight)>=0){
      loaded = true;
      capacity -=weight;
      System.out.println("Cargo " + contentID + " loaded for transport");
    }else{
      System.out.println("Cannot load cargo, exceeds weight capacity");
    }
  }
  public void unload(){
    loaded = false;
    capacity+=weight;
    System.out.println("Cargo " + contentID + " unloaded");
  }
  public static double capacity(){
    return capacity;
  }
  public void details(){
    System.out.print("Cargo ID: " + contentID + ", ");
    System.out.print("Contents: " + contents + ", ");
    System.out.print("Weight: " + weight + ", ");
    System.out.println("Loaded: " + loaded);
  }
}
