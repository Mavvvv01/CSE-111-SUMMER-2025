public class Book extends Product{
  public String isbn;
  public String publisher;
  
  public Book(int n, String t, int p, String i, String pub){
    super(n,t,p);
    isbn = i;
    publisher = pub;
  }
  
  public String printDetail(){
    return getIdTitlePrice() + " " + "ISBN: " + isbn + " Publisher: " + publisher;
  }
}

public class CD extends Product{
  public String band;
  public int dur;
  public String gen;
  
  public CD(int n, String t, int p, String b, int d, String g){
    super(n, t, p);
    band = b;
    dur = d;
    gen = g;
  }
  public String printDetail(){
    return getIdTitlePrice() + " " + "Band: " + band + " Duration: " + dur + " minutes " + "Genre: " + gen;
  }
}

public class ComplexNumber extends RealNumber{
  public double imagValue;
  public ComplexNumber(){
    super(1.0);
    imagValue = 1.0;
  }
  public ComplexNumber(double r, double i){
    super(r);
    imagValue = i;
  }
  public String toString(){
    return super.toString() + "\n" + "Imaginary Part: " + imagValue;
  }
}
